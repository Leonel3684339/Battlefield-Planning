Measures local elevation variability (standard deviation within a moving window). High roughness indicates difficult terrain that might channel movement.

python
def compute_roughness(elev_grid: np.ndarray, window_size: int = 3) -> np.ndarray:
    from scipy.ndimage import generic_filter
    def std_filter(values):
        return np.std(values)
    return generic_filter(elev_grid, std_filter, size=window_size)
